// .github/workflows/blank.yml
name: YouTube Downloader
on:
  workflow_dispatch:
    inputs:
      video_url:
        description: 'YouTube URL'
        required: true
      quality_profile:
        description: 'Quality Profile'
        required: true
        default: 'mp4_standard'

jobs:
  download:
    runs-on: ubuntu-latest
    steps:
      - name: Install yt-dlp
        run: |
          sudo curl -L https://github.com/yt-dlp/yt-dlp/releases/latest/download/yt-dlp -o /usr/local/bin/yt-dlp
          sudo chmod a+rx /usr/local/bin/yt-dlp

      - name: Download Process
        env:
          COOKIES: ${{ secrets.YOUTUBE_COOKIES }}
        run: |
          echo "$COOKIES" > cookies.txt
          
          case "${{ github.event.inputs.quality_profile }}" in
            "high_video")
              yt-dlp -f "bestvideo+bestaudio/best" --merge-output-format mp4 --cookies cookies.txt "${{ github.event.inputs.video_url }}" -o "downloaded_file.mp4"
              ;;
            "mp4_standard")
              yt-dlp -f "bestvideo[ext=mp4]+bestaudio[ext=m4a]/best[ext=mp4]/best" --cookies cookies.txt "${{ github.event.inputs.video_url }}" -o "downloaded_file.mp4"
              ;;
            "fast_video")
              yt-dlp -f "worstvideo+worstaudio/worst" --cookies cookies.txt "${{ github.event.inputs.video_url }}" -o "downloaded_file.mp4"
              ;;
            "high_mp3")
              yt-dlp -x --audio-format mp3 --audio-quality 0 --cookies cookies.txt "${{ github.event.inputs.video_url }}" -o "downloaded_file.mp3"
              ;;
            "fast_mp3")
              yt-dlp -x --audio-format mp3 --audio-quality 9 --cookies cookies.txt "${{ github.event.inputs.video_url }}" -o "downloaded_file.mp3"
              ;;
          esac

      - name: Upload Result
        uses: actions/upload-artifact@v4
        with:
          name: youtube-download
          path: downloaded_file.*
